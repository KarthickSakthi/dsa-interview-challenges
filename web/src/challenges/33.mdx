### [Search in Rotated Sorted Array](https://leetcode.com/problems/search-in-rotated-sorted-array/)

- Use binary search technique to find the index of rotation (Challenge [153](#find-minimum-in-rotated-sorted-array))
- Apply the normal binary search with the rotation index as offset

```js copy
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
  const minIdx = findMinIdx(nums);
  let left = 0, right = nums.length - 1;

  while (left <= right) {
    const mid = left + Math.floor((right - left) / 2);
    const realMid = (mid + minIdx) % nums.length;

    if (nums[realMid] === target) {
      return realMid;
    } else if (target < nums[realMid]) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }

  return -1;
};

var findMinIdx = function (nums) {
  let left = 0, right = nums.length - 1;
  let minIdx = 0;

  while (left <= right) {
    if (nums[left] < nums[right]) {
      return nums[minIdx] < nums[left] ? minIdx : left;
    }

    const mid = left + Math.floor((right - left) / 2);
    if (nums[left] <= nums[mid]) {
      minIdx = nums[minIdx] < nums[left] ? minIdx : left;
      left = mid + 1;
    } else {
      minIdx = nums[minIdx] < nums[mid] ? minIdx : mid;
      right = mid - 1;
    }
  }

  return minIdx;
};
```

##### Complexity

- Time: O(log(n))
- Space: O(1)
